📖 کتابخانه Rubkey

Rubkey یک فریمورک مدرن و قدرتمند پایتون برای ساخت ربات های پیام رسان روبیکا است.

چرا Rubkey
  • آماده - نصب از طریق pip و آغاز استفاده در کمترین زمان
  • ساده - ساختاری تمیز، قابل فهم و مناسب برای مبتدیان
  • سریع - استفاده از فناوری های مدرن و کدنویسی بهینه
  • پشتیبانی از Async - طراحی همزمان و غیرهمزمان
  • قدرتمند - ارائه تمام قابلیت های API ربات روبیکا

🧰 نصب و راه اندازی

pip install rubkey
import rubkey
print(rubkey.__version__)  # 2.0.7

⚡ تفاوت Sync و Async

# Sync - ساده
from rubkey import Rubkey

bot = Rubkey(token="TOKEN")

@bot.command("start")
def start(msg):
    msg.reply("سلام")

# Async - سریعتر
import asyncio
from rubkey import AsyncBot

bot = AsyncBot(token="TOKEN")

@bot.command("start")
async def start(msg):
    await msg.reply("سلام")

async def main():
    await bot.run()

asyncio.run(main())

🎛️ کیپدها

Chat Keypad (کیپد معمولی)

from rubkey import Rubkey, Keypad, KeyButton

bot = Rubkey(token="TOKEN")

keypad = Keypad(resize=True, one_time=False)
keypad.add_row(
    KeyButton("btn1", "راهنما"),
    KeyButton("btn2", "درباره")
)
keypad.add_row(KeyButton("btn3", "تماس", button_type="AskMyPhoneNumber"))

@bot.command("start")
def start(msg):
    msg.reply("خوش آمدید", keypad=keypad)

@bot.on_button("btn1")
def btn1(msg):
    msg.reply("راهنما")

@bot.on_button("btn2")
def btn2(msg):
    msg.reply("درباره")

bot.run()

📁 ارسال فایل (SendFile)

import asyncio
from rubkey import AsyncBot
from rubkey.async_rubkey import AsyncSendFile

bot = AsyncBot(token="TOKEN")

@bot.command("send")
async def send(msg):
    sendfile = AsyncSendFile(bot)
    result = await sendfile.request_send_file("Image")
    upload_url = result["upload_url"]
    upload_result = await sendfile.upload_file(upload_url, "/path/to/file.png")
    file_id = upload_result["data"]["file_id"]
    await sendfile.send_file(msg.chat_id, file_id, "عکس شما")

async def main():
    await bot.run()

asyncio.run(main())

🚫 بن و آنبن در گروه

import asyncio
from rubkey import AsyncBot

bot = AsyncBot(token="TOKEN")

@bot.command("ban")
async def ban(msg):
    if not msg.chat_id.startswith("g"):
        return
    target = bot.message_map.get(msg.reply_to_message_id)
    if target:
        await bot.ban_chat_member(msg.chat_id, target)
        await msg.reply("کاربر بن شد")

@bot.command("unban")
async def unban(msg):
    if not msg.chat_id.startswith("g"):
        return
    target = bot.message_map.get(msg.reply_to_message_id)
    if target:
        await bot.unban_chat_member(msg.chat_id, target)
        await msg.reply("کاربر آنبن شد")

async def main():
    await bot.run()

asyncio.run(main())

🔍 فیلترها

from rubkey import Rubkey, Filters

bot = Rubkey(token="TOKEN")

@bot.command("start")
def start(msg):
    msg.reply("سلام")

@bot.on_message()
def echo(msg):
    if msg.chat_id.startswith("b") and msg.text and not msg.text.startswith("/"):
        msg.reply(msg.text)

bot.run()
📌 فیلترهای موجود
  • Filters.chat_type("private") - فیلتر پیوی
  • Filters.chat_type("group") - فیلتر گروه
  • Filters.text("متن") - فیلتر متن دقیق
  • Filters.regex(r"...") - فیلتر regex
  • Filters.has_file() - فیلتر وجود فایل

📱 SendCodeButton

دکمه تکست باکس با کیبورد عددی برای ارسال کد تایید.

from rubkey import Rubkey, Keypad
from rubkey.selfbot import SendCodeButton

bot = Rubkey(token="TOKEN")
send_code_btn = SendCodeButton("send_code", "ارسال کد", place_holder="شماره...", title="شماره تلفن")

@bot.command("start")
def start(msg):
    keypad = Keypad().add_row(send_code_btn)
    msg.reply("شماره تلفن:", keypad=keypad)

@bot.on_button("send_code")
def send_code(msg):
    send_code_btn.handle(bot, msg.chat_id, msg.text)

bot.run()

🔗 Middleware و Router

from rubkey import Rubkey, Router

bot = Rubkey(token="TOKEN")

def log_middleware(msg, bot):
    print(f"پیام از {msg.sender_id}")
    return msg

bot.add_middleware(log_middleware)

router = Router()

@router.command("test")
def test(msg):
    msg.reply("Router")

bot.include_router(router)

@bot.command("start")
def start(msg):
    msg.reply("سلام")

bot.run()

💾 Session و Database

from rubkey import Rubkey, Session

bot = Rubkey(token="TOKEN")
session = Session("user_data.json")

@bot.command("save")
def save(msg):
    session.set(msg.sender_id, "name", "تست")
    msg.reply("ذخیره شد")

@bot.command("load")
def load(msg):
    name = session.get(msg.sender_id, "name", "ناشناس")
    msg.reply(f"نام: {name}")

bot.run()

import asyncio

bot = AsyncGroupBot(token="TOKEN")
